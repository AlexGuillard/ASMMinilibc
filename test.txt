strchr:
    XOR RAX, RAX
    MOV RAX, 0

    loop:
        INC RDI
        CMP BYTE[RDI], 0
        JE end_of_string
        CMP [RDI], RSI
        JNE loop

    end_of_string:
        MOV RAX, RDI
        MOV RSP, RBP
        POP RBP
        RET

strchr:
    ENTER 0, 0
    XOR RAX, RAX

    loop:
        CMP BYTE[RDI], 0
        JE RET
        INC RDI
        CMP [RDI], RSI
        JNE loop

    MOV RAX, RDI
    LEAVE
    RET
